## Nolga teng bo'lmagan ikkita son berilgan. Ularning yig'indisi, ko'paytmasi va har birining kvadrati aniqlansin.
```cpp
#include <iostream>
#include <math.h>
using namespace std;
int main()
{
    float a, b, P, S, a2, b2;
    cout << "a = ";cin >> a;
    cout << "b = ";cin >> b;
    P = a + b;
    S = a * b;
    a2 = a * a;
    b2 = b * b;
    cout << "a + b = " << P << endl;
    cout << "a * b = " << S << endl;
    cout << "a * a = " << a2 << endl;
    cout << "b * b = " << b2;
}
```
